# StudyListPagination

## Import

```javascript
import { StudyListPagination } from '@ohif/ui';
```

## Basic usage

<Playground>
  {() => {
    const [currentPage, setCurrentPage] = useState(1);
    const [perPage, setPerPage] = useState(25);
    const paginationData = {
      onChangePage: (page) => {
        setCurrentPage(page);
      },
      onChangePerPage: (perPage) => {
        setPerPage(perPage);
        setCurrentPage(1);
      },
      currentPage,
      perPage,
    };
    return <StudyListPagination paginationData={paginationData} />;
  }}
</Playground>
